233 // Build the 'operators' structure from Biogeme 'Operator' Python object
234 void setOperators (PyObject* pOperator) {
235 
236  int i;
237  int len=sizeof(listOfOperatorName)/sizeof(listOfOperatorName[0]);
238  PyObject* pOperatorValue;
239  PyObject* pOpIndex;
240 
241  operators.minUnopIndex = PyLong_AsLong(PyObject_GetAttrString(pOperator, MIN_UNOP_INDEX));
242  operators.maxUnopIndex = PyLong_AsLong(PyObject_GetAttrString(pOperator, MAX_UNOP_INDEX));
243  operators.minBinopIndex = PyLong_AsLong(PyObject_GetAttrString(pOperator, MIN_BINOP_INDEX));
244  operators.maxBinopIndex = PyLong_AsLong(PyObject_GetAttrString(pOperator, MAX_BINOP_INDEX));
245  operators.minIteratoropIndex = PyLong_AsLong(PyObject_GetAttrString(pOperator, MIN_ITERATOROP_INDEX));
246  operators.maxIteratoropIndex = PyLong_AsLong(PyObject_GetAttrString(pOperator, MAX_ITERATOROP_INDEX));
247  operators.undefOp = PyLong_AsLong(PyObject_GetAttrString(pOperator, UNDEF_OP));
248 
249  for (i=0; i<len; ++i) {
250  pOperatorValue = PyObject_GetAttrString(pOperator, const_cast<char *>(listOfOperatorName[i].c_str()));
251  patString theOperator(PyBytes_AsString(PyUnicode_AsASCIIString(pOperatorValue))) ;
252  char format[] = "s" ;
253  char op[] = GET_OPERATOR_FUNC ;
254  pOpIndex = PyObject_CallMethod(pOperator, op, format, theOperator.c_str());
255  operators.indexByNames[listOfOperatorName[i]] = PyLong_AsLong(pOpIndex);
256  }
257 }
