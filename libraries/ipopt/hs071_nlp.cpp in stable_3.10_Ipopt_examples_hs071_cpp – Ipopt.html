47// returns the variable bounds
48bool HS071_NLP::get_bounds_info(Index n, Number* x_l, Number* x_u,
49                                Index m, Number* g_l, Number* g_u)
50{
51  // here, the n and m we gave IPOPT in get_nlp_info are passed back to us.
52  // If desired, we could assert to make sure they are what we think they are.
53  assert(n == 4);
54  assert(m == 2);
55
56  // the variables have lower bounds of 1
57  for (Index i=0; i<4; i++) {
58    x_l[i] = 1.0;
59  }
60
61  // the variables have upper bounds of 5
62  for (Index i=0; i<4; i++) {
63    x_u[i] = 5.0;
64  }
65
66  // the first constraint g1 has a lower bound of 25
67  g_l[0] = 25;
68  // the first constraint g1 has NO upper bound, here we set it to 2e19.
69  // Ipopt interprets any number greater than nlp_upper_bound_inf as
70  // infinity. The default value of nlp_upper_bound_inf and nlp_lower_bound_inf
71  // is 1e19 and can be changed through ipopt options.
72  g_u[0] = 2e19;
73
74  // the second constraint g2 is an equality constraint, so we set the
75  // upper and lower bound to the same value
76  g_l[1] = g_u[1] = 40.0;
77
78  return true;
79}

115{

127bool HS071_NLP::eval_g(Index n, const Number* x, bool new_x, Index m, Number* g)
128{

181//return the structure or values of the hessian
182bool HS071_NLP::eval_h(Index n, const Number* x, bool new_x,
183                       Number obj_factor, Index m, const Number* lambda,
184                       bool new_lambda, Index nele_hess, Index* iRow,
185                       Index* jCol, Number* values)
186{
187  if (values == NULL) {
188    // return the structure. This is a symmetric matrix, fill the lower left
189    // triangle only.
190
191    // the hessian for this problem is actually dense
192    Index idx=0;
193    for (Index row = 0; row < 4; row++) {
194      for (Index col = 0; col <= row; col++) {
195        iRow[idx] = row;
196        jCol[idx] = col;
197        idx++;
198      }
199    }
200
201    assert(idx == nele_hess);
202  }
203  else {
204    // return the values. This is a symmetric matrix, fill the lower left
205    // triangle only
206
207    // fill the objective portion
208    values[0] = obj_factor * (2*x[3]); // 0,0
209
210    values[1] = obj_factor * (x[3]);   // 1,0
211    values[2] = 0.;                    // 1,1
212
213    values[3] = obj_factor * (x[3]);   // 2,0
214    values[4] = 0.;                    // 2,1
215    values[5] = 0.;                    // 2,2
216
217    values[6] = obj_factor * (2*x[0] + x[1] + x[2]); // 3,0
218    values[7] = obj_factor * (x[0]);                 // 3,1
219    values[8] = obj_factor * (x[0]);                 // 3,2
220    values[9] = 0.;                                  // 3,3
221
222
223    // add the portion for the first constraint
224    values[1] += lambda[0] * (x[2] * x[3]); // 1,0
225
226    values[3] += lambda[0] * (x[1] * x[3]); // 2,0
227    values[4] += lambda[0] * (x[0] * x[3]); // 2,1
228
229    values[6] += lambda[0] * (x[1] * x[2]); // 3,0
230    values[7] += lambda[0] * (x[0] * x[2]); // 3,1
231    values[8] += lambda[0] * (x[0] * x[1]); // 3,2
232
233    // add the portion for the second constraint
234    values[0] += lambda[1] * 2; // 0,0
235
236    values[2] += lambda[1] * 2; // 1,1
237
238    values[5] += lambda[1] * 2; // 2,2
239
240    values[9] += lambda[1] * 2; // 3,3
241  }
242
243  return true;
244}
